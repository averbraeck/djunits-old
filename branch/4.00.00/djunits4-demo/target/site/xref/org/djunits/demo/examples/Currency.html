25  public final class Currency implements UNITS
26  {
27      /** */
28      private Currency()
29      {
30          // utility constructor.
31      }
32  
33      /**
34       * Create some scalar values to demonstrate conversion from and to related units.
35       * @param args String[]; the command line arguments; not used
36       */
37      public static void main(final String[] args)
38      {
39          Locale.setDefault(Locale.US); // Ensure that floating point values are printed using a dot (".")
40          MoneyUnit.setStandardUnit(MoneyUnit.EUR);
41          Density density = new Density(150.0, KG_PER_METER_3);
42          MoneyPerMass costPerKG = new MoneyPerMass(20.0, MoneyPerMassUnit.EUR_PER_KILOGRAM);
43          System.out.println("standard price " + costPerKG); // prints 20.000ï¿½/kg
44          Money startupCost = new Money(100.0, MoneyUnit.EUR);
45          Volume volume = new Volume(300.0, LITER);
46          Mass totalWeight = volume.multiplyBy(density);
47          System.out.println("weight of " + volume + " is " + totalWeight); // prints 45.000kg
48          Money totalCost = startupCost.plus(totalWeight.multiplyBy(costPerKG));
49          System.out.println("total cost " + totalCost); // prints 1000.000EUR
50      }
51  }
