17  public final class AbsAndRel
18  {
19      /** */
20      private AbsAndRel()
21      {
22          //
23      }
24  
25      /**
26       * @param args String[]; args
27       */
28      public static void main(final String[] args)
29      {
30          AbsoluteTemperature t = new AbsoluteTemperature(0.0, AbsoluteTemperatureUnit.DEGREE_FAHRENHEIT);
31          System.out.println("Temperature t  = " + t + ", si = " + t.getSI());
32          System.out.println("t in Kelvin    = " + t.toString(AbsoluteTemperatureUnit.KELVIN));
33          System.out.println("t in Celsius   = " + t.toString(AbsoluteTemperatureUnit.DEGREE_CELSIUS));
34  
35          // add 32 degrees Fahrenheit - should be 0 Celsius
36          System.out.println("\nadd 32 degrees Fahrenheit - should be 0 Celsius");
37          Temperature t32 = new Temperature(32.0, TemperatureUnit.DEGREE_FAHRENHEIT);
38          AbsoluteTemperature t2 = t.plus(t32);
39          System.out.println("Temperature t2 = " + t2);
40          System.out.println("t2 in Kelvin   = " + t2.toString(AbsoluteTemperatureUnit.KELVIN));
41          System.out.println("t2 in Celsius  = " + t2.toString(AbsoluteTemperatureUnit.DEGREE_CELSIUS));
42  
43          // The other way around to add 32 degrees Fahrenheit should also be 0 Celsius
44          System.out.println("\nThe other way around to add 32 degrees Fahrenheit should also be 0 Celsius");
45          AbsoluteTemperature t3 = t32.plus(t);
46          System.out.println("Temperature t3 = " + t3);
47          System.out.println("t3 in Kelvin   = " + t3.toString(AbsoluteTemperatureUnit.KELVIN));
48          System.out.println("t3 in Celsius  = " + t3.toString(AbsoluteTemperatureUnit.DEGREE_CELSIUS));
49  
50          // Subtraction - should yield a difference of 32 degrees Fahrenheit (17.78 K/Celcius)
51          System.out.println("\nSubtraction - should yield a difference of 32 degrees Fahrenheit (17.78 K/Celcius)");
52          Temperature t4 = t3.minus(t);
53          System.out.println("Temperature t4 = " + t4);
54          System.out.println("t4 in Kelvin   = " + t4.toString(TemperatureUnit.KELVIN));
55          System.out.println("t4 in Celsius  = " + t4.toString(TemperatureUnit.DEGREE_CELSIUS));
56  
57      }
