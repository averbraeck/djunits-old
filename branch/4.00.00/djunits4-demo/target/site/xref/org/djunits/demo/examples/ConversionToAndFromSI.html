19  public final class ConversionToAndFromSI implements UNITS
20  {
21      /** */
22      private ConversionToAndFromSI()
23      {
24          // utility constructor.
25      }
26  
27      /**
28       * Create some scalar values to demonstrate conversion from and to related units.
29       * @param args String[]; the command line arguments; not used
30       */
31      public static void main(final String[] args)
32      {
33          Locale.setDefault(Locale.US); // Ensure that floating point values are printed using a dot (".")
34          Length length = new Length(123, KILOMETER); // Construct a Relative Length
35          System.out.println("length is " + length); // prints 123.000km; i.e. uses original unit
36          System.out.println("length in METER is " + length.toString(METER)); // prints 1.230+05m
37          System.out.println("si value is " + length.si); // prints 123000.0
38          Length delta = new Length(250, MILE);
39          System.out.println("delta is " + delta); // prints 250.000mi
40          System.out.println("length + delta is " + length.plus(delta)); // prints 5.253e+05m
41          System.out.println("length + delta in km is " + (length.plus(delta)).toString(KILOMETER)); // prints 525.336km
42      }
43  }
