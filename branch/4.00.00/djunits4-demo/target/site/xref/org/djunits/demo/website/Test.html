27  public final class Test
28  {
29      /** Furlong example. */
30      public static final LengthUnit FURLONG = new LengthUnit("Furlong", "fr", UnitSystem.IMPERIAL, LengthUnit.FOOT, 660);
31  
32      /** Fortnight example. */
33      public static final DurationUnit FORTNIGHT = new DurationUnit("Fortnight", "fn", UnitSystem.OTHER, DurationUnit.DAY, 14);
34  
35      /** fr/fn example. */
36      public static final SpeedUnit FURLONGS_PER_FORTNIGHT =
37              new SpeedUnit(FURLONG, FORTNIGHT, "Furlongs per Fortnight", "fr/fn", UnitSystem.OTHER);
38  
39      /** */
40      private Test()
41      {
42          // utility class
43      }
44  
45      /**
46       * @param args String[]; args
47       * @throws ValueException on error
48       */
49      public static void main(final String[] args) throws ValueException
50      {
51          Length fr1000 = new Length(1000.0, FURLONG);
52          Duration twoWeeks = new Duration(1.0, FORTNIGHT);
53          Speed speed = fr1000.divideBy(twoWeeks);
54          System.out.println(speed);
55          System.out.println(speed.toString(FURLONGS_PER_FORTNIGHT));
56          System.out.println();
57  
58          DoubleScalar.Rel<JerkUnit> jerk1 = new DoubleScalar.Rel<>(1.2, JerkUnit.SI);
59          System.out.println("jerk1 = new DoubleScalar.Rel<>(1.2, JerkUnit.SI) : " + jerk1);
60          DoubleScalar.Rel<JerkUnit> jerk2 = jerk1.multiplyBy(2.0);
61          System.out.println("jerk2 = jerk1.multiplyBy(2.0)                    : " + jerk2);
62          double[] sv = new double[] {1, 2, 3, 4, 5};
63          DoubleVector.Rel<JerkUnit> jerkVector = new DoubleVector.Rel<JerkUnit>(sv, JerkUnit.SI, StorageType.DENSE);
64          System.out.println("jerkVector: : " + jerkVector);
65  
66          Jerk jjerk1 = new Jerk(1.2, JerkUnit.SI);
67          System.out.println("jerk1 = new Jerk(1.2, JerkUnit.SI) : " + jjerk1);
68          Jerk jjerk2 = jjerk1.multiplyBy(2.0);
69          System.out.println("jerk2 = jerk1.multiplyBy(2.0)      : " + jjerk2);
70  
71          double[][] data = new double[1000][1000];
72          for (int i = 0; i < 1000; i++)
73          {
74              for (int j = 0; j < 1000; j++)
75              {
76                  data[i][j] = 9 * i + 2 * j * 0.364;
77              }
78          }
79          MutableLengthMatrix lengthMatrix = new MutableLengthMatrix(data, LengthUnit.CENTIMETER, StorageType.DENSE);
80          lengthMatrix.round();
81      }
